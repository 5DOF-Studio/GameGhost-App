// --- ScreenCaptureKit native helper (GaimerScreenCapture.xcframework) ---

/// <summary>Returns true if ScreenCaptureKit is available (macCatalyst 18.2+ / macOS 14+).</summary>
[DllImport("GaimerScreenCapture", CallingConvention = CallingConvention.Cdecl)]
[return: MarshalAs(UnmanagedType.U1)]
internal static extern bool sck_is_available();

/// <summary>
/// Captures a window via ScreenCaptureKit. Calls back with PNG bytes (data, length)
/// or (null, 0) on failure. The callback fires on the main thread.
/// </summary>
[DllImport("GaimerScreenCapture", CallingConvention = CallingConvention.Cdecl)]
internal static extern void sck_capture_window(
    uint windowID, int width, int height,
    SckCaptureCallback callback);
```

**DllImport library name:** Use `"GaimerScreenCapture"` (the framework name, NOT "libGaimerScreenCapture" -- .NET resolves this against the linked xcframework).

**Important:** The `SckCaptureCallback` delegate must be pinned by the caller (kept alive via GCHandle or local variable) for the duration of the native call. Document this in a comment.

Do NOT modify any existing P/Invoke declarations or enums. Only ADD the new SCK-related declarations.

    try

    }
}
```
Cache this at class load because it will not change at runtime. Wrap in try/catch for resilience -- if the native library fails to load, fall back gracefully.

2. **Add CaptureWithScreenCaptureKit method:**
```csharp
/// <summary>
/// Captures a window using ScreenCaptureKit via the native helper.
/// Returns PNG bytes, or empty array on failure/timeout.
/// </summary>
private static byte[] CaptureWithScreenCaptureKit(CaptureTarget target)
{
    var windowId = (uint)target.Handle;
    var width = (int)target.BoundsWidth;
    var height = (int)target.BoundsHeight;

    // Use 2x for Retina (SCK captures at pixel resolution)
    width = Math.Max(width * 2, 100);
    height = Math.Max(height * 2, 100);

    var tcs = new TaskCompletionSource<byte[]>(TaskCreationOptions.RunContinuationsAsynchronously);

    // The delegate must stay alive for the duration of the native call.
    // Prevent GC collection by holding a strong reference.
    NativeMethods.SckCaptureCallback callback = (dataPtr, length) =>
    {
        if (dataPtr == IntPtr.Zero || length <= 0)
        {
            tcs.TrySetResult(Array.Empty<byte>());
            return;
        }

        var buffer = new byte[length];
        System.Runtime.InteropServices.Marshal.Copy(dataPtr, buffer, 0, length);
        tcs.TrySetResult(buffer);
    };

    // Pin the delegate to prevent GC during native call
    var handle = System.Runtime.InteropServices.GCHandle.Alloc(callback);
    try
    {
        NativeMethods.sck_capture_window(windowId, width, height, callback);

        // Block with timeout -- SCK capture should complete within seconds
        if (!tcs.Task.Wait(TimeSpan.FromSeconds(10)))
        {
            System.Diagnostics.Debug.WriteLine("[Capture] SCK capture timed out");
            return Array.Empty<byte>();
        }

        return tcs.Task.Result;
    }
    catch (Exception ex)
    {
        System.Diagnostics.Debug.WriteLine($"[Capture] SCK error: {ex.Message}");
        return Array.Empty<byte>();
    }
    finally
    {
        handle.Free();
    }
}
```
